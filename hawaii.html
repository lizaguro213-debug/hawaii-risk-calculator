    <script>

        function formatMoney(num) {

            return '$' + num.toLocaleString('en-US');

        }

 

        function updateBudget(val) {

            document.getElementById('budgetDisplay').textContent = formatMoney(parseInt(val));

        }

 

        function updateYears(val) {

            document.getElementById('yearsDisplay').textContent = val + ' лет';

        }

 

        function toggleLeasehold() {

            const ownership = document.getElementById('ownership').value;

            const yearsDiv = document.getElementById('leaseholdYears');

            if (ownership === 'lease') {

                yearsDiv.style.display = 'block';

            } else {

                yearsDiv.style.display = 'none';

            }

        }

 

        function calculate() {

            const budget = parseInt(document.getElementById('budget').value);

            const ownership = document.getElementById('ownership').value;

            const years = parseInt(document.getElementById('years').value);

            const floodZone = document.getElementById('floodZone').value;

            const needRental = document.getElementById('rental').checked;

            const needInsurance = document.getElementById('insurance').checked;

 

            let riskScore = 0;

            let warnings = [];

            let costs = [];

 

            // 1. Тип владения

            if (ownership === 'lease') {

                if (

 

years < 30) {

                    riskScore += 40;

                    warnings.push({type: 'red', text: '⚠️ Осталось менее 30 лет — банк не даст ипотеку, ликвидность нулевая'});

                    costs.push(['Дисконт ликвидности', budget * 0.30]);

                } else if (years < 50) {

                    riskScore += 25;

                    warnings.push({type: 'yellow', text: '⚠️ Осталось 30–50 лет — сложная ипотека, снижение стоимости при перепродаже'});

                } else {

                    riskScore += 10;

                    warnings.push({type: 'green', text: 'ℹ️ Аренда >50 лет — приемлемо, но проверьте условия пересмотра платы'});

                }

            } else {

                warnings.push({type: 'green', text: '✅ Fee Simple — минимальный риск по владению'});

            }

 

            // 2. Зона наводнения

            if (floodZone === 've') {

                riskScore += 35;

                warnings.push({type: 'red', text: '🚨 VE-зона: страховка обязательна, стоимость $8–22K/год, эрозия неизбежна'});

                costs.push(['Страхование HPIA (VE)', 15000]);

            } else if (floodZone === 'ae') {

                riskScore += 25;

                warnings.push({type: 'yellow', text: '⚠️ AE-зона: страховка $4–12K/год, возможны ограничения на ремонт'});

                costs.push(['Страхование HPIA (AE)', 8000]);

            } else {

                riskScore += 5;

                warnings.push({type: 'green', text: '✅ X-зона: страховка опциональна, ~$2–4K/год'});

                costs.push(['Страхование (X)', 3000]);

            }

 

            // 3. Аренда

            if (needRental) {

                riskScore += 20;

                warnings.push({type: 'red', text: '🚫 Краткосрочная аренда запрещена в жилых зонах Kauai (мораторий с 2008). Только долгосрочная или личное проживание'});

                costs.push(['Упущенная выгода аренды/мес', 8000]);

            }

 

            // 4. Страхование урагана

            if (needInsurance && floodZone === 'x') {

                costs.push(['Страхование ураган', 5000]);

            }

 

            // 5. Обязательные проверки

            costs.push(['Термит-инспекция', 350]);

            costs.push(['Title insurance', budget * 0.005]);

            costs.push(['Юрист / transaction coordinator', 3500]);

 

            // Результаты

            const totalHidden = costs.reduce((sum, c) => sum + c[1], 0);

            const pct = (totalHidden / budget) * 100;

 

            // Уровень риска

            let riskText, riskClass;

            if (riskScore <= 30) {

                riskText = '🟢 Низкий';

                riskClass = 'risk-low';

            } else if (riskScore <= 60) {

                riskText = '🟡 Средний';

                riskClass = 'risk-medium';

            } else {

                riskText = '🔴 Высокий';

                riskClass = 'risk-high';

            }

 

            /

 

/ Отображение

            document.getElementById('riskLevel').innerHTML = '<span class="' + riskClass + '">' + riskText + '</span><br><small>' + riskScore + ' баллов</small>';

            document.getElementById('hiddenCosts').textContent = formatMoney(Math.round(totalHidden));

            document.getElementById('pctBudget').textContent = pct.toFixed(1) + '%';

 

            // Предупреждения

            const warningsDiv = document.getElementById('warnings');

            warningsDiv.innerHTML = warnings.map(w =>

                '<div class="warning-item warning-' + w.type + '">' + w.text + '</div>'

            ).join('');

 

            // Таблица расходов

            const tbody = document.getElementById('costTable');

            let tableHTML = '';

            costs.forEach(c => {

                tableHTML += '<tr><td>' + c[0] + '</td><td>' + formatMoney(Math.round(c[1])) + '</td></tr>';

            });

            tableHTML += '<tr class="total"><td>ИТОГО</td><td>' + formatMoney(Math.round(totalHidden)) + '</td></tr>';

            tbody.innerHTML = tableHTML;

 

            // Рекомендации

            const recDiv = document.getElementById('recommendation');

            let recHTML = '';

            if (riskScore > 60) {

                recDiv.className = 'recommendation rec-red';

                recHTML = '<h3>❌ Не рекомендуется без дополнительной due diligence</h3><ul>' +

                    '<li>Получить insurance quote (HPIA или частный страховщик)</li>' +

                    '<li>Проверить Shoreline Certification и SMA Assessment</li>' +

                    '<li>Заказать геотехническую экспертизу (оползни)</li>' +

                    '<li>Проверить TVNC статус (если планируется аренда)</li>' +

                    '<li>Нанять local counsel (Kauai attorney, не риэлтора)</li>' +

                    '</ul>';

            } else if (riskScore > 30) {

                recDiv.className = 'recommendation rec-yellow';

                recHTML = '<h3>⚠️ Покупка возможна при выполнении условий</h3><ul>' +

                    '<li>Title Commitment с проверкой leasehold (если применимо)</li>' +

                    '<li>Независимая termite inspection</li>' +

                    '<li>Проверка HOA financials и CC&Rs</li>' +

                    '<li>Insurance contingency в договоре</li>' +

                    '</ul>';

            } else {

                recDiv.className = 'recommendation rec-green';

                recHTML = '<h3>✅ Риски управляемы</h3><ul>' +

                    '<li>Escrow agent + title insurance</li>' +

                    '<li>Home inspection ($400–600)</li>' +

                    '<li>Юрист на review договора ($2–5K)</li>' +

                    '</ul>';

            }

            recDiv.innerHTML = recHTML;

 

            // Показать результаты

            document.getElementById('results').classList.add('show');

        }
